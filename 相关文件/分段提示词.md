### 学习页面布局示例

```typescript:src/pages/study/index.vue
<template>
  <view class="study-container">
    <!-- 顶部分类选择器 -->
    <category-selector
      :categories="categories"
      :current="currentCategory"
      @change="handleCategoryChange"
    />
    
    <!-- 题目展示区 -->
    <question-card
      :question="currentQuestion"
      @view-detail="handleViewDetail"
    />
    
    <!-- 底部控制栏 -->
    <view class="control-bar">
      <button @click="handlePrev">上一题</button>
      <button @click="handleFavorite">
        {{ isFavorite ? '取消收藏' : '收藏' }}
      </button>
      <button @click="handleNext">下一题</button>
    </view>
  </view>
</template>
```
